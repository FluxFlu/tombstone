Purity

Purity:

A pure variable is a variable that is unable to be modified. A pure function is a function that does not modify the values within it. Example: 


// add doesn't need to be impure, because it doesn't modify any values.
fn add = i32 (i32 x, i32 y) => {
    return x + y;
}

// sum_array is impure because it includes `i++`. Loops are also considered impure because they cannot be used from a pure context.
impure fn sum_array = i32 (i32* x) => {
    i32 acc = 0;
    for(i32 i = 0; i < 3; i++) {
        acc += x[i]; 
    }
    return acc;
}

// The following code doesn't run, because it's a pure function that attempts to call an impure function (sum_array).
export fn main = i32 (i32 x) => {
    i32* arr = i32 [1, 2, 3];
    return sum_array(arr);
}
